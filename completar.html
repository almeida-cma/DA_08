public class DesafioDefault {
    String nome;
    int idade;

    // CONSTRUTOR DEFAULT

package tema08;

public class DesafioParametrizado {
    String modelo;
    int ano;

📌 Sobrecarga de Construtores

📚 CONCEITO: Vários construtores com diferentes parâmetros.
🎯 OBJETIVO: Oferecer diferentes formas de criar objetos.
⚡ MÉTODO: Múltiplos construtores com assinaturas diferentes.
📌 Complete as lacunas para criar sobrecarga de construtores 
package tema08;

public class DesafioSobrecarga {
    String produto;
    double preco;
    int quantidade;

    // CONSTRUTOR 1: default
    public () {
        this.produto = "Nao informado";
        this.preco = 0.0;
        this.quantidade = 0;
    }

    // CONSTRUTOR 2: com produto e preco
    public (String produto, double preco) {
        .produto = produto;
        .preco = preco;
        this.quantidade = 0;
    }

    // CONSTRUTOR 3: com produto, preco e quantidade
    public DesafioSobrecarga(String produto, double preco, int quantidade) {
        this.produto = produto;
        .preco = preco;
        .quantidade = quantidade;
    }

    public void exibirInfo() {
        System.out.println("Produto: " + produto);
        System.out.println("Preco: R$ " + preco);
        System.out.println("Quantidade: " + quantidade);
    }

    public static void main(String[] args) {
        System.out.println("--- CONSTRUTOR 1 (default) ---");
        DesafioSobrecarga p1 = new DesafioSobrecarga();
        p1.exibirInfo();

        System.out.println("\n--- CONSTRUTOR 2 (produto e preco) ---");
        DesafioSobrecarga p2 =  DesafioSobrecarga("Mouse", 50.0);
        p2.exibirInfo();

        System.out.println("\n--- CONSTRUTOR 3 (produto, preco e quantidade) ---");
        DesafioSobrecarga p3 = new DesafioSobrecarga("Teclado", 100.0, 5);
        p3.();
    }
}

📌 this() para chamar outro construtor

📚 CONCEITO: this() chama outro construtor da mesma classe.
🎯 OBJETIVO: Reaproveitar código entre construtores.
⚡ MÉTODO: this(parametros); deve ser a primeira linha.
📌 Complete as lacunas para usar this() entre construtores 
package tema08;

public class DesafioThis {
    String nome;
    int idade;
    String cidade;

    // CONSTRUTOR PRINCIPAL (com todos os parâmetros)
    public DesafioThis(String nome, int idade, String cidade) {
        .nome = nome;
        .idade = idade;
        .cidade = cidade;
    }

    // CONSTRUTOR com nome e idade (chama o principal com cidade padrão)
    public DesafioThis(String nome, int idade) {
        (nome, idade, "Nao informada");
    }

    // CONSTRUTOR só com nome (chama o principal com idade e cidade padrão)
    public DesafioThis(String nome) {
        (nome, 0, "Nao informada");
    }

    public void exibirDados() {
        System.out.println("Nome: " + nome);
        System.out.println("Idade: " + idade);
        System.out.println("Cidade: " + cidade);
    }

    public static void main(String[] args) {
        System.out.println("--- PESSOA 1 (só nome) ---");
        DesafioThis p1 = new DesafioThis("Joao");
        p1.();

        System.out.println("\n--- PESSOA 2 (nome e idade) ---");
        DesafioThis p2 = new ("Maria", 25);
        p2.exibirDados();

        System.out.println("\n--- PESSOA 3 (nome, idade e cidade) ---");
        DesafioThis p3 =  DesafioThis("Pedro", 30, "Sao Paulo");
        p3.exibirDados();
    }
}
